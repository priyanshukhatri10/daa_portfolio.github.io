C++ Code for Heap Sort on Delivery Schedule

This is the C++ code that uses the Heap Sort algorithm to sort a delivery schedule by urgency or distance.

#include <iostream>
#include <vector>
using namespace std;

void heapify(vector<pair<int, string>> &arr, int n, int i) {
    int largest = i;
    int left = 2 * i + 1;
    int right = 2 * i + 2;

    if (left < n && arr[left].first > arr[largest].first)
        largest = left;

    if (right < n && arr[right].first > arr[largest].first)
        largest = right;

    if (largest != i) {
        swap(arr[i], arr[largest]);
        heapify(arr, n, largest);
    }
}

void heapSort(vector<pair<int, string>> &arr) {
    int n = arr.size();

    for (int i = n / 2 - 1; i >= 0; i--)
        heapify(arr, n, i);

    for (int i = n - 1; i > 0; i--) {
        swap(arr[0], arr[i]);
        heapify(arr, i, 0);
    }
}

void printSchedule(const vector<pair<int, string>> &arr) {
    cout << "Sorted Delivery Schedule (by urgency/distance):" << endl;
    for (const auto &entry : arr) {
        cout << "Urgency/Distance: " << entry.first << ", Delivery: " << entry.second << endl;
    }
}

int main() {
    vector<pair<int, string>> deliverySchedule = {
        {5, "Package A"}, {3, "Package B"}, {8, "Package C"},
        {6, "Package D"}, {2, "Package E"}
    };

    cout << "Original Delivery Schedule:" << endl;
    for (const auto &entry : deliverySchedule) {
        cout << "Urgency/Distance: " << entry.first << ", Delivery: " << entry.second << endl;
    }

    heapSort(deliverySchedule);

    printSchedule(deliverySchedule);
    return 0;
}
