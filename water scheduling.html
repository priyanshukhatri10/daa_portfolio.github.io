C++ Water Usage Scheduling


#include <iostream>
#include <vector>
#include <algorithm>  // For max() function

using namespace std;

void scheduleWaterUsage(const vector<int>& waterDemand, int windowSize, int totalWater) {
    int n = waterDemand.size();
    int currentWater = 0;
    int maxWaterUsage = 0;

    for (int i = 0; i < windowSize; ++i) {
        currentWater += waterDemand[i];  // Add water demand in the window
    }

    maxWaterUsage = currentWater;  // Initialize max usage with the first window
    for (int i = windowSize; i < n; ++i) {
        currentWater = currentWater - waterDemand[i - windowSize] + waterDemand[i];
        maxWaterUsage = max(maxWaterUsage, currentWater);
    }

    cout << "Maximum water usage in any " << windowSize << "-hour window: " << maxWaterUsage << endl;

    if (maxWaterUsage <= totalWater) {
        cout << "Water usage is within limits." << endl;
    } else {
        cout << "Water usage exceeds the available supply. Adjusting allocation..." << endl;
    }
}

int main() {
    vector<int> waterDemand = {10, 15, 20, 30, 25, 40, 35, 50, 45, 30, 20, 15, 10, 5, 15, 25, 35, 40, 45, 50, 40, 35, 30, 20};

    int windowSize = 6;  
    int totalWater = 200;  

    scheduleWaterUsage(waterDemand, windowSize, totalWater);

    return 0;
}
