C++ Code for Industrial Development in Shyamora

This is the C++ code that implements the Divide and Conquer approach for industrial development in Shyamora.

#include <iostream>
#include <vector>
#include <string>
using namespace std;


struct Zone {
    string name;
    string type;
    int area; 
    bool infrastructureComplete;
    bool resourcesAllocated;
    
    Zone(string n, string t, int a, bool infrastructure, bool resources)
        : name(n), type(t), area(a), infrastructureComplete(infrastructure), resourcesAllocated(resources) {}
};


Zone getZoneInfo() {
    string name, type;
    int area;
    char infrastructureComplete, resourcesAllocated;
    
    /
    cout << "Enter Zone Name: ";
    getline(cin, name);
    cout << "Enter Zone Type (e.g., Heavy Industry, Light Industry, Commercial): ";
    getline(cin, type);
    cout << "Enter Zone Area (in square kilometers): ";
    cin >> area;
    cin.ignore();  
    cout << "Is infrastructure complete? (y/n): ";
    cin >> infrastructureComplete;
    cout << "Are resources allocated? (y/n): ";
    cin >> resourcesAllocated;
    cin.ignore(); 
    
  
    bool infrastructure = (infrastructureComplete == 'y' || infrastructureComplete == 'Y');
    bool resources = (resourcesAllocated == 'y' || resourcesAllocated == 'Y');
    
   
    return Zone(name, type, area, infrastructure, resources);
}


void developZone(Zone &zone) {
  
    if (!zone.infrastructureComplete) {
        zone.infrastructureComplete = true;
        cout << zone.name << " Infrastructure is completed." << endl;
    }
    
 
    if (!zone.resourcesAllocated) {
        zone.resourcesAllocated = true;
        cout << zone.name << " Resources have been allocated." << endl;
    }
}


void developIndustrialArea(vector<Zone> &zones) {
   
    for (auto &zone : zones) {
        cout << "\nDeveloping " << zone.name << " (" << zone.type << ") with area: " << zone.area << " km²" << endl;
        developZone(zone); // Conquer each zone
    }
}


void printDevelopmentStatus(const vector<Zone> &zones) {
    cout << "\nFinal Development Status:\n";
    for (const auto &zone : zones) {
        cout << "Zone: " << zone.name << "\nType: " << zone.type
             << "\nArea: " << zone.area << " km²"
             << "\nInfrastructure Completed: " << (zone.infrastructureComplete ? "Yes" : "No")
             << "\nResources Allocated: " << (zone.resourcesAllocated ? "Yes" : "No") << endl;
    }
}

int main() {
    int numZones;
    
   
    cout << "Enter the number of zones to develop: ";
    cin >> numZones;
    cin.ignore(); 

    vector<Zone> industrialZones;

    
    for (int i = 0; i < numZones; ++i) {
        cout << "\nEnter details for Zone " << (i + 1) << ":\n";
        industrialZones.push_back(getZoneInfo());
    }

   
    developIndustrialArea(industrialZones);

   
    printDevelopmentStatus(industrialZones);

    return 0;
}
